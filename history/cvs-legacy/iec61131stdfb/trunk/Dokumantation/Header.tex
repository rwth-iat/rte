%	jpg-Abbildung mit Bildunterschrift (Caption)
%		#1: Dateiname des Bildes
%		#2: Ausrichtung
%		#3: gew端nschte Breite
%		#4: Referenz
%		#5: Bildunterschrift
%		#6: Anordnung auf der Seite (optionales "p")
%
\newcommand{\jpgCap}[6]{%
	\begin{figure}[htb#6]%
		\begin{#2}%
			\includegraphics[width=#3]{Bilder/#1.jpg}%
		\end{#2}%
		\caption{#5}%
		\label{fig:#4}%
	\end{figure}%
}%

\newcommand{\pdfCap}[6]{%
	\begin{figure}[htb#6]%
		\begin{#2}%
			\includegraphics[width=#3]{Bilder/#1.pdf}%
		\end{#2}%
		\caption{#5}%
		\label{fig:#4}%
	\end{figure}%
}%
